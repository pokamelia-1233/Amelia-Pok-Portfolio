Time_IDs = unique(stock$time_id)[1:500]

log_r_stock <- list()

for (i in 1:length(Time_IDs)) {
  seconds = stock %>% filter(time_id == Time_IDs[i]) %>%
    pull(seconds_in_bucket)
  price = stock %>% filter(time_id == Time_IDs[i]) %>% 
    pull(WAP)
  log_r = log(price[-1] / price[1:length(price) - 1]) 
  log_r_stock[[i]] = tibble(time = seconds[-1], log_return = log_r)
  
  no_diff <- (1:600)[!(1:600 %in% log_r_stock[[i]]$time)]
  if (length(no_diff) >0){
    no_diff_df = tibble(time = no_diff, log_return = 0)
    log_r_stock[[i]] = log_r_stock[[i]] %>% 
      bind_rows(no_diff_df) %>%
      arrange(time)
  }
    
    
}
